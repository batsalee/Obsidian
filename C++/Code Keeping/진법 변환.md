## 2. C++ style Code

```C++
#include <iostream>
#include <bitset>

void digit2bitset()
{
    /* 1. 10진수 정수 -> 2진수 */

    // bitset 사용방식
    int n2bit = 5;
    std::cout << "5는 2진수로 : " << std::bitset<8 * sizeof(n2bit)>(n2bit) << std::endl;

    // 직접 구현 방식
    int result = 0;
    for (int i = 1; n2bit >= 1; i *= 10) {
        result = (n2bit % 2) * i + result;
        n2bit /= 2;
    }
    std::cout << "5는 2진수로 : " << result << std::endl;

    /* 2. 10진수 실수 -> 2진수 */
    // 소수점 윗부분 윗쪽의 코드처럼 처리하면 되므로 아래 코드는 소수점 아래 부분 처리 코드만 작성    
    double d2bit = 0.625;
    std::cout << "0.625는 2진수로 0.";
    while (d2bit != 0) {
        d2bit *= 2;
        std::cout << (int)d2bit;

        if (d2bit >= 1) d2bit -= 1;
    }
    std::cout << std::endl;

    /* 3. 2진수 -> 10진수 */

    // bitset 사용방식
    // bitset 객체에 넣고 to_ulong() 하면 끝
    int num1 = 10;
    int num2 = 15;

    std::bitset<32> b1(num1);  // b1 = 1010
    std::bitset<32> b2(num2);  // b2 = 1111

    std::cout << b1 << std::endl; // 출력시 000000.....1010
    std::cout << b1.to_ulong() << std::endl; // 출력시 10진수 10

    // 직접 구현 방식
    result = 0;
    int bit = 100; // 2진수 100이니 10진수로 4
    for (int j = 0; bit >= 1; j++) {
        result += (bit % 10) * pow(2, j);
        bit /= 10;
    }
    std::cout << "이진수 100은 10진수로 " << result << std::endl;
}
```
